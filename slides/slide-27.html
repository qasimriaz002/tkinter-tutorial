Login Application Example

A login system demonstrates conditional logic, input retrieval, and screen navigation.

Workflow:

  1. User enters credentials.
  2. Click "Login" triggers validation.
  3. If Valid: Hide login frame, show dashboard.
  4. If Invalid: Show messagebox.showerror.

import tkinter as tk from tkinter import messagebox def login(): u = user_entry.get() p = pass_entry.get() if u == "admin" and p == "1234": login_frame.pack_forget() dashboard_frame.pack(fill="both", expand=True) tk.Label(dashboard_frame, text="Welcome!", font=("Arial", 20)).pack(expand=True) else: messagebox.showerror("Error", "Wrong credentials") root = tk.Tk() login_frame = tk.Frame(root) login_frame.pack(padx=20, pady=20) user_entry = tk.Entry(login_frame) pass_entry = tk.Entry(login_frame, show="*") user_entry.pack(); pass_entry.pack() tk.Button(login_frame, text="Login", command=login).pack() dashboard_frame = tk.Frame(root) root.mainloop()
